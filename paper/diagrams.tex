\newcommand{\AIONDiagramForkAndWormhole}{

  ]

  % Fork at next tick

  \draw[fork] (u2) -- (u3p) node[midway, right] {\scriptsize $\mu_2'$};

  \node[font=\scriptsize] at (3,1.35) {shared prefix};
